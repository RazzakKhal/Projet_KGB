CREATE TABLE Agent (
CODE INT NOT NULL PRIMARY KEY,
NOM VARCHAR(36) NOT NULL,
PRENOM VARCHAR(36) NOT NULL,
DATE INT NOT NULL,
NATIONALITE VARCHAR(36) NOT NULL,
SPECIALITE VARCHAR(36) NOT NULL
);

CREATE TABLE Contact(
ID INT NOT NULL PRIMARY KEY,
NOM VARCHAR(36) NOT NULL,
PRENOM VARCHAR(36) NOT NULL,
DATE INT NOT NULL,
NOMCODE VARCHAR(36) NOT NULL UNIQUE,
NATIONALITE VARCHAR(36) NOT NULL
);

CREATE TABLE Cible(
ID INT NOT NULL PRIMARY KEY,
NOM VARCHAR(36) NOT NULL,
PRENOM VARCHAR(36) NOT NULL,
DATE INT NOT NULL,
NOMCODE VARCHAR(36) NOT NULL UNIQUE,
NATIONALITE VARCHAR(36) NOT NULL
);

CREATE TABLE Planque(
CODE INT NOT NULL PRIMARY KEY,
ADRESSE VARCHAR(36) NOT NULL,
PAYS VARCHAR(36) NOT NULL,
TYPE VARCHAR(36) NOT NULL

);

CREATE TABLE Mission (
ID INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
TITRE VARCHAR(36) NOT NULL,
DESCRIPTION CHAR NOT NULL,
NOMCODE VARCHAR (36) NOT NULL UNIQUE,
PAYS VARCHAR(36) NOT NULL,
AGENT INT NOT NULL,
CONTACT INT NOT NULL,
CIBLE INT NOT NULL,
TYPE VARCHAR(36) NOT NULL,
STATUT VARCHAR(36) NOT NULL,
PLANQUE INT NULL,
DATEDEBUT INT NOT NULL,
DATEFIN INT NOT NULL,
SPECIALITE VARCHAR(36) NOT NULL,
FOREIGN KEY (AGENT) REFERENCES Agent(CODE),
FOREIGN KEY (PLANQUE) REFERENCES Planque(CODE),
FOREIGN KEY (CONTACT) REFERENCES Contact(ID),
FOREIGN KEY (CIBLE) REFERENCES Cible(ID)
);

CREATE TABLE Admin(
ID INT NOT NULL PRIMARY KEY,
NOM VARCHAR(36) NOT NULL,
PRENOM VARCHAR(36)NOT NULL,
MAIL VARCHAR(255) NOT NULL,
PASS VARCHAR(60) NOT NULL,
DATE INT NOT NULL
);
